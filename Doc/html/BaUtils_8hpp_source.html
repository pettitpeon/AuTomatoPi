88 static inline uint64_t DirSize(
89  std::string path,
90  uint32_t maxDepth = (uint32_t)-1
91  ) {
92 
93  // When maximum depth is reached, stop
94  if (maxDepth == 0) {
95  return 0;
96  }
97 
98  DIR *d;
99  struct dirent *de; // directory entry
100  uint64_t total = 0;
101  struct stat info;
102 #ifdef _WIN32
103  char del = '\\';
104 #else
105  char del = '/';
106 #endif
107 
108  d = opendir(path.c_str());
109  if (!d) {
110  return 0;
111  }
112 
113  std::string file;
114  for (de = readdir(d); de != 0; de = readdir(d)) {
115 
116  // Skip "." and ".."
117  if ((strcmp(de->d_name, "..") == 0) || (strcmp(de->d_name, ".") == 0)) {
118  continue;
119  }
120 
121  // Get the file info
122  file = path + de->d_name;
123  if(stat( file.c_str(), &info ) == 0) {
124 
125  // Check if is directory and recurse
126  if( info.st_mode & S_IFDIR ) {
127  total += DirSize(file + del, --maxDepth);
128  } else {
129  total += info.st_size;
130  }
131  }
132  }
133 
134  closedir(d);
135  return total;
136 }
