117 static inline uint64_t DirSize(
118  std::string path,
119  uint32_t maxDepth = (uint32_t)-1
120  ) {
121 
122  // When maximum depth is reached, stop
123  if (maxDepth == 0) {
124  return 0;
125  }
126 
127  DIR *d;
128  struct dirent *de; // directory entry
129  uint64_t total = 0;
130  struct stat info;
131 #ifdef _WIN32
132  char del = '\\';
133 #else
134  char del = '/';
135 #endif
136 
137  d = opendir(path.c_str());
138  if (!d) {
139  return 0;
140  }
141 
142  std::string file;
143  for (de = readdir(d); de != 0; de = readdir(d)) {
144 
145  // Skip "." and ".."
146  if ((strcmp(de->d_name, "..") == 0) || (strcmp(de->d_name, ".") == 0)) {
147  continue;
148  }
149 
150  // Get the file info
151  file = path + de->d_name;
152  if(stat( file.c_str(), &info ) == 0) {
153 
154  // Check if is directory and recurse
155  if( info.st_mode & S_IFDIR ) {
156  total += DirSize(file + del, --maxDepth);
157  } else {
158  total += info.st_size;
159  }
160  }
161  }
162 
163  closedir(d);
164  return total;
165 }

390  if (!pIs) {

392  }
393 
394  bool err;

397  pError = &err;
398  }
399 
400  // Check file size
401  pIs->seekg(0, std::ios::end);
402  std::string::size_type size = pIs->tellg();
403 
404  if (size == std::string::npos) {
405  *pError = true;
406  return "";
407  }
408 
409  // Resize string to size
410  s.resize(size);
411 
412  // Rewind stream
413  pIs->seekg(0, std::ios::beg);
414 
415  // Read contents to string
416  pIs->read(&s[0], s.size());
417  *pError = false;
418

420  pIs->clear();
421  pIs->seekg(0, std::ios::beg);
422
